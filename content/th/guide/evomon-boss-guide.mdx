  <Step num="1" title="วิเคราะห์รูปแบบบอส">

  </Step>
  <Step num="2" title="จัดลำดับความสำคัญของการกำหนดเป้าหมาย">

  </Step>
  <Step num="3" title="กำหนดจังหวะบัฟ/ดีบัฟของคุณ">

  </Step>
  <Step num="4" title="ใช้สกิลสร้างความเสียหายสูง">

  </Step>
  <Step num="5" title="จัดการการรักษาและการล้างสถานะ">

  </Step>
</Steps>

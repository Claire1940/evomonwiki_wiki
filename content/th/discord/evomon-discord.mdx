  <Step num="2" title="ค้นหาโค้ดที่ใช้งานได้">

  </Step>
  <Step num="3" title="เริ่มเกม Evomon ใน Roblox">

  </Step>
  <Step num="4" title="เข้าสู่เมนูการตั้งค่า">

  <Step num="5" title="วางและแลกรับ">

  </Step>
</Steps>
